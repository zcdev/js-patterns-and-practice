** start of script.js **

const markdownInput = document.querySelector("#markdown-input"); const htmlOutput = document.querySelector("#html-output"); const preview = document.querySelector("#preview");

function convertMarkdown() {
  let text = markdownInput.value;

  // Headings
  text = text.replace(/^\s*###\s+(.*)$/gm, "<h3>$1</h3>");
  text = text.replace(/^\s*##\s+(.*)$/gm, "<h2>$1</h2>");
  text = text.replace(/^\s*#\s+(.*)$/gm, "<h1>$1</h1>");

  // Images
  text = text.replace(
    /!\[([^\]]+)\]\(([^)]+)\)/g,
    '<img alt="$1" src="$2">'
  );

  // Links
  text = text.replace(
    /\[([^\]]+)\]\(([^)]+)\)/g,
    '<a href="$2">$1</a>'
  );

  // Bold
  text = text.replace(
    /(\*\*|__)(.*?)\1/g,
    "<strong>$2</strong>"
  );

  // Italic
  text = text.replace(
    /(\*|_)(.*?)\1/g,
    "<em>$2</em>"
  );

  // Blockquotes
  text = text.replace(
    /^\s*>\s+(.*)$/gm,
    "<blockquote>$1</blockquote>"
  );

  return text;
}

markdownInput.addEventListener("input", () => {
  const html = convertMarkdown();
  htmlOutput.textContent = html;
  preview.innerHTML = html;
});
